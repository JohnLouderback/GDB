var model={
    firstName: "Agatha",
    lastName: "Trunchbull",
    fullName: {
        in: function(){
            var lastName=model.lastName;
            if(lastName.length>0)
                lastName=" "+lastName;
            else
                lastName="";
            return model.firstName+lastName;
        },
        out: function(value){
            var nameArr=value.split(" ");
            model.firstName=nameArr[0];
            if(nameArr.length>1)
                model.lastName=nameArr[1];
            else
                model.lastName="";
        }
    }
};
GDB({teacher: model});
